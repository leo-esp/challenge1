import { Meta, ColorPalette, ColorItem } from "@storybook/addon-docs";
import { theme } from "./colors";

<Meta title="Colors" />

<ColorPalette>
  {Object.keys(theme).map((colorKey) => {
    if (colorKey === "type") {
      return;
    }
    if (colorKey === "neutrals") {
      const neutrals = Object.keys(theme["neutrals"]).reduce(
        (total, currentValue) => ({
          ...total,
          [currentValue]: theme[colorKey][currentValue],
        }),
        {}
      );
      return (
        <ColorItem
          title={colorKey}
          subtitle={`theme.${colorKey}`}
          colors={neutrals}
        />
      );
    }
    return (
      <ColorItem
        title={colorKey}
        subtitle={`theme.${colorKey}`}
        colors={{ [colorKey]: theme[colorKey] }}
      />
    );
  })}
</ColorPalette>
